# Metrics Evaluation - RMSE, RASE & Adjusted R-squared Score

def calculate_rmse(y_true, y_pred):
    mse = mean_squared_error(y_true, y_pred)
    rmse = np.sqrt(mse)
    return rmse

def calculate_rase(y_true, y_pred):
    numerator = np.sum(np.square(y_true - y_pred))
    denominator = np.sum(np.square(y_true - np.mean(y_true)))
    rase = numerator / denominator
    return rase

def calculate_adjustedR2(r2, n, p):
    adjusted_r2 = 1 - ((1 - r2) * (n - 1)) / (n - p - 1)
    return adjusted_r2
